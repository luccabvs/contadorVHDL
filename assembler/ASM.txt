!SETUP
LDI $0 --Carrega 0 no acumulador
STA @0 --Valor 0 em unidade
STA @1 --Valor 0 em dezena
STA @2 --Valor 0 em centena
STA @3 --Valor 0 em unidade de milhar
STA @4 --Valor 0 em dezena de milhar
STA @5 --Valor 0 em centena de milhar
STA @6 --Valor 0 em limite de unidade
STA @7 --Valor 0 em limite de dezena
STA @8 --Valor 0 em limite de centena
STA @9 --Valor 0 em limite de unidade de milhar
STA @10 --Valor 0 em limite de dezena de milhar
STA @11 --Valor 0 em limite de centena de milhar
STA @256 --Valor 0 no conjunto de 8 LEDs
STA @257 --Valor 0 no LEDR8
STA @258 --Valor 0 no LEDR9
STA @288 --Valor 0 no HEX0
STA @289 --Valor 0 no HEX1
STA @290 --Valor 0 no HEX2
STA @291 --Valor 0 no HEX3
STA @292 --Valor 0 no HEX4
STA @293 --Valor 0 no HEX5
STA @510 --Limpa leitura KEY1
STA @511 --Limpa leitura KEY0
STA @509 --Limpa leitura FPGA RESET
STA @61 --Armazena o valor 0 em MEM[61]
LDI $1 --Carrega 1 no Acumulador
STA @62 --Armazena o valor 1 em MEM[62]
LDI $10 --Carregado 10 no Acumulador
STA @63 --Armazena o valor 1 em MEM[63]
JSR @INCUNIDADE --Chama a função INCUNIDADE
JSR @DISPLAY --Chama a função DISPLAY
!INCUNIDADE
STA @511 --Limpa leitura KEY0
LDA @0 --Carrega o valor das unidades
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @INCDEZENA --Se for igual a 10, incrementa a dezena
STA @0 --Armazena o valor do acumulador em HEX0
RET
!INCDEZENA
LDA @61 --Carrega o valor de 0 no acumulador
STA @0 --Zera as unidades
LDA @1 --Carrega o valor das dezenas
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @INCCENTENA --Se for igual a 10, incrementa a centena
STA @1 --Armazena o valor do acumulador em HEX1
RET
!INCCENTENA
LDA @61 --Carrega o valor de 0 no acumulador
STA @1 --Zera as dezenas
LDA @2 --Carrega o valor das centenas
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @INCUMILHAR --Se for igual a 10, incrementa a unidade de milhar
STA @2 --Armazena o valor do acumulador em HEX2
RET
!INCUMILHAR
LDA @61 --Carrega o valor de 0 no acumulador
STA @2 --Zera as centenas
LDA @3 --Carrega o valor das unidades de milhar
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @INCDEZMILHAR --Se for igual a 10, incrementa a dezena de milhar
STA @3 --Armazena o valor do acumulador em HEX3
RET
!INCDEZMILHAR
LDA @61 --Carrega o valor de 0 no acumulador
STA @3 --Zera as unidades de milhar
LDA @4 --Carrega o valor das dezenas de milhar
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @INCCENTMILHAR --Se for igual a 10, incrementa a centena de milhar
STA @4 --Armazena o valor do acumulador em HEX4
RET
!INCCENTMILHAR
LDA @61 --Carrega o valor de 0 no acumulador
STA @4 --Zera as dezenas de milhar
LDA @5 --Carrega o valor das centenas de milhar
SOMA @63 --Soma o valor do acumulador com 1
CEQ @62 --Compara o valor do acumulador com 10
JEQ @LIMITE --Se for igual a 10, incrementa a centena de milhar
STA @5 --Armazena o valor do acumulador em HEX5
RET
!DISPLAY
LDA @0 --Carrega o valor das unidades
STA @288 --Armazena o valor do acumulador em HEX0
LDA @1 --Carrega o valor das dezenas
STA @289 --Armazena o valor do acumulador em HEX1
LDA @2 --Carrega o valor das centenas
STA @290 --Armazena o valor do acumulador em HEX2
LDA @3 --Carrega o valor das unidades de milhar
STA @291 --Armazena o valor do acumulador em HEX3
LDA @4 --Carrega o valor das dezenas de milhar
STA @292 --Armazena o valor do acumulador em HEX4
LDA @5 --Carrega o valor das centenas de milhar
STA @293 --Armazena o valor do acumulador em HEX5
RET
